00_hello

hello_spec.rb 

#hello_spec.rb

require "hello"

describe "the hello function" do
  it "says hello" do
    hello.should == "Hello!"
  end
end

describe "the greet function" do
  it "says hello to someone" do
    greet("Alice").should == "Hello, Alice!"
  end

  it "says hello to someone else" do
    greet("Bob").should == "Hello, Bob!"
  end
end



hello.rb 

#hello.rb

def hello
    "Hello!"
end

def greet(someone)
    "Hello, #{someone}!"    
end
